private
  variable
    n : Nat
    γ : Conₘ n
    Γ : Con Term n

Unitʳ : ⊢ Γ
      → ∃ λ ([Γ] : ⊩ᵛ Γ)
      → ∃ λ ([U] : Γ ⊩ᵛ⟨ ¹ ⟩ U / [Γ])
      → γ ▸ Γ ⊩ʳ⟨ ¹ ⟩ Unit ∷ U / [Γ] / [U]
Unitʳ ⊢Γ =
  let [Γ] = valid ⊢Γ
      [U] = Uᵛ [Γ]
  in  [Γ] , [U] , λ [σ] x → Uᵣ (Unitⱼ ε)

starʳ : ∀ {l} → ⊢ Γ
      → ∃ λ ([Γ] : ⊩ᵛ Γ)
      → ∃ λ ([Unit] : Γ ⊩ᵛ⟨ l ⟩ Unit / [Γ])
      → γ ▸ Γ ⊩ʳ⟨ l ⟩ star ∷ Unit / [Γ] / [Unit]
starʳ ⊢Γ =
  let [Γ] = valid ⊢Γ
      [Unit] = Unitᵛ [Γ]
  in  [Γ] , [Unit] , λ [σ] x → starᵣ (starⱼ ε) T.refl
